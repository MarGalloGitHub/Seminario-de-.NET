void Fac(int n , out int resultado)
{
    int i;
    resultado = 1;
    for (i = 1 ; i <= n ; i++)
    {
        resultado = (resultado * i);
    }
}




void Fac2(int n , out int resultado)
{
    if (n == 0 || n == 1)
    {
       resultado = 1;
       
    }
    else
    {
        Fac2(n-1, out resultado);
        resultado*= n;
    }
}
